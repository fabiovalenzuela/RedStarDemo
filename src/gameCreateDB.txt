CREATE TABLE Monster(iD int Primary Key not Null, name text not null, description text not null, hitPoints int not null, minDamage int not null, maxDamage int not null, chanceHit int not null);
CREATE TABLE Player(iD int Primary Key not Null, name text not null, description text not null, hitPoints int not null, minDamage int not null, maxDamage int not null, chanceHit int not null);
CREATE TABLE Character(iD int Primary Key not Null, name text not null, description text not null, hitPoints int not null, minDamage int not null, maxDamage int not null, chanceHit int not null);
CREATE TABLE Exit(exitID int Primary Key not Null, direction text not null, destination int not null);
CREATE TABLE ExitRoom(roomID int Primary Key not Null, exitID int not null);
CREATE TABLE ItemRoom(roomID int Primary Key not Null, itemID int not null);
CREATE TABLE MonsterRoom(roomID int Primary Key not Null, iD int not null);
CREATE TABLE Item(itemID int Primary Key not Null, name text not null, description text not null, itemRoomID int not null);
CREATE TABLE Puzzle(puzzleID int Primary Key not Null, puzzleName text not null, puzzleDesc text not null, puzzleType, puzzleRoomID int not null);
